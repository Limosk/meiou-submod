Timur_Vassalise = {
	if = {
		limit = {
			NOT = { exists = $tag$ }
			any_owned_province = { is_core = $tag$ }
		}
		
		hidden_effect = {
			every_owned_province = {
				limit = {
					is_core = $tag$
				}
				remove_core = PREV
				add_claim = PREV
				cede_and_release = { scope = $tag$ }
			}
			$tag$ = { country_event = { id = Init.2 } }
		}
		create_subject = {
			subject_type = $subject_type$
			subject = $tag$
		}
	}
	else = {
		hidden_effect = {
			every_owned_province = {
				limit = {
					is_core = $tag$
				}
				remove_core = PREV
				add_claim = PREV
				cede_and_release = { scope = $tag$ }
			}
		}
	}
}
