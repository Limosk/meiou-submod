### Second Peace of Thorn by Vorondil ###


country_event = {			#Poland version
	id = secondpeaceofthorn.1
	title = "secondpeaceofthorn.1.t"
	desc = "secondpeaceofthorn.1.d"
	picture = SIEGE_eventPicture
  
	fire_only_once = yes
  
	trigger = {
		tag = POL
		NOT = { exists = PLC }
		has_country_flag = first_peace_of_thorn
		has_country_flag = supported_prussian_confederation_flag
		NOT = { has_country_flag = refused_peace_of_thorn }
		TEU = {
				NOT = { 
				is_at_war = yes
			}
			is_free_or_tributary_trigger = yes
			any_owned_province = {
					region = prussia_region
			}
			NOT = { num_of_cities = 10 }
		}
		OR = { 
			41 = {
						owned_by = POL
				}
			41 = {
						owned_by = DNZ
				}
			1844 = {
						owned_by = POL
				}
			1844 = {
						owned_by = DNZ
				}
		}
	}
  
	mean_time_to_happen = { months = 3 }
  
	option = {
		name = "secondpeaceofthorn.1.a"
		every_owned_province = {
			limit = {
					region = prussia_region
					owned_by = ROOT
					NOT = { province_id = 2398 } #Chelmno
			}
			cede_province = TEU
			add_core = TEU
		}
		every_province = {
			limit = {
					region = prussia_region
					owned_by = DNZ
					NOT = { province_id = 2398 } #Chelmno
				NOT = { province_id = 2355 } #Danzig
			}
			cede_province = TEU
			add_core = TEU
		}
		every_province = {
			limit = {
				area = mazovia_area
				owned_by = TEU
			}
			cede_province = ROOT
		}
		every_province = {
			limit = {
				area = lithuania_area
				owned_by = TEU
			}
			cede_province = LIT
		}
		if = {
			limit = {
				WRM = {
					exists = yes
					is_subject_of = TEU
				}
			}
			every_province = {
				limit = {
					region = prussia_region
					owned_by = WRM
					NOT = { province_id = 2398 } #Chelmno
					NOT = { province_id = 2355 } #Danzig
				}
				cede_province = TEU
				add_core = TEU
			}
		}
		if = {
			limit = {
				2398 = {
					owned_by = TEU
				}
			}
			2398 = {
				cede_province = POL
				add_core = POL
			}
		}
		
		TEU = {
			free_vassal = LVO
		}
		
		LIT = {
			if = {
					limit = {
						is_subject_of = POL
					is_subject_of_type = permanent_union
					}
					every_owned_province = {
						limit = {
							region = prussia_region
							owned_by = THIS
							NOT = { province_id = 2398 } #Chelmno
						}
						remove_core = THIS
						cede_province = TEU
						add_core = TEU
					}
			}
		}
		country_event = {
			id = secondpeaceofthorn.2
		}
	}
}
