country_event = {
	id = haiti.2
	title = "haiti.EVTNAME2"
	desc = "haiti.EVTDESC2"
	picture = COLONIZATION_eventPicture
	major = yes
	trigger = {
		is_year = 1700
		technology_group = western
		num_of_owned_provinces_with = {
			value = 2
			region = carribeans_region
			culture = maroon
		}
		is_colonial_nation = no
		NOT = {
			capital_scope = {
				region = carribeans_region
			}
		}
		NOT = {
			exists = HAT
		}
	}
	mean_time_to_happen = {
		years = 50
		modifier = {
			factor = 0.8
			culture_group = langue_d_oil
		}
		modifier = {
			factor = 0.8
			NOT = {
				accepted_culture = maroon
			}
		}
		modifier = {
			factor = 0.8
			NOT = {
				stability = 1
			}
		}
		modifier = {
			factor = 0.8
			owns = 488
		}
		modifier = {
			factor = 0.8
			owns = 1536
		}
		modifier = {
			factor = 0.8
			owns = 1537
		}
	}
	immediate = {
		hidden_effect = {
			every_owned_province = {
				limit = {
					region = carribeans_region
					culture = maroon
				}
				discover_country = HAT
				add_core = HAT
			}
			release = HAT
			HAT = {
				Pol_ChangeGovRepublic = { Mandate=2 Stylings=1 ExtraPowers=1 Assembly=3 TermOfOffice=3 } 
				add_government_reform = Pol_Nobility4
			}

		}
	}
	option = {
		name = "haiti.EVTOPTA2"
		HAT = {
			declare_war_with_cb = {
				who = ROOT
				casus_belli = cb_independence_war
			}
		}
		hidden_effect = {
			HAT = {
				country_event = {
					id = haiti.3
					days = 1
				}
			}
		}
	}
}

country_event = {
    id = haiti.4
    title = "haiti.EVTNAME4"
    desc = "haiti.EVTDESC4"
    picture = COLONIZATION_eventPicture
    is_triggered_only = yes
    trigger = {
        technology_group = western
        carribeans_region = {
            owned_by = PREV
            NOT = { 
                culture = maroon
            }
        }
        NOT = {
            has_country_modifier = the_abolish_slavery_act
        }
    }
    mean_time_to_happen = {
        days = 1
    }
    option = {
		name = "haiti.EVTOPTA4"
        random_owned_province = {
            limit = {
                region = carribeans_region
            }
            change_culture = maroon
            add_unrest = 5
            set_province_flag = afro_carribean
        }
    }
}
