no_cavalry = {
	potential = {
		has_country_flag = no_cavalry
	}
	
	trigger = {
		always = yes
	}
	
	shock_damage_received = 0.25
}
silkroad = {
	potential = {
		NOT = { dip_tech = 10 }
		capital_scope = {
			province_group = steppes_group
		}
	}
	
	trigger = {
		always = yes
	}
	
	trade_range_modifier = 0.75
}
transsahara = {
	potential = {
		NOT = { dip_tech = 15 }
		capital_scope = {
			OR = {
				region = west_sahara_region
				region = east_sahara_region
			}
		}
	}
	
	trigger = {
		always = yes
	}
	
	trade_range_modifier = 1.0
}

africa_trade = {
	potential = {
		NOT = { dip_tech = 10 }
		capital_scope = {
			OR = {
				continent = africa
				region = west_sahara_region
				region = east_sahara_region
			}
		}
	}
	
	trigger = {
		always = yes
	}
	
	trade_range_modifier = 0.75
}
