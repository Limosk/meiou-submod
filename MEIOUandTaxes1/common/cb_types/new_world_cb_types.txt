	is_triggered_only = yes # See cb_types.1
	months = 12000
	ai_peace_desire = -30

	prerequisites = {
		OR = {
			AND = {
				has_idea = exploration_ideas_6
	
				NOT = {
					capital_scope = {
						OR = {
							continent = south_america
							continent = north_america
						}
					}
				}
			}
			AND = {
				is_colonial_nation = yes
				overlord = {
					NOT = {
						capital_scope = {
							OR = {
								continent = south_america
								continent = north_america
							}
						}
					}
				}
			}
		}
		
		FROM = {
			capital_scope = {
				OR = {
					continent = south_america
					continent = north_america
				}
			}
			
			is_colonial_nation = no
			
			NOT = {
				has_institution = Banking
			}
		}
		
		is_revolutionary = no
	}

	war_goal = new_world_conquest
}
