# flavor_hyw.14 - France is crushed
country_event = {
	id = flavor_hyw.14
	title = "flavor_hyw.11.t"
	desc = "flavor_hyw.14.d"
	picture = DIPLOMACY_eventPicture
	
	trigger = {
		tag = FRA
		war_with = ENG
		183 = { controlled_by = ENG } #Paris
		has_global_flag = hundred_years_war
		NOT = {
			any_owned_province = {
				region = east_france_region
				controlled_by = ROOT
			}
		}
		num_of_owned_provinces_with = { value = 1 OR = { region = aquitaine_region region = languedoc_region region = provence_region } }
		any_owned_province = {
			OR = { region = aquitaine_region region = languedoc_region region = provence_region }
			controlled_by = ROOT
		}
	}
	
	mean_time_to_happen = { months = 6 }
	
	immediate = {
		clr_global_flag = hundred_years_war
		#_#_log = "FE-TempHYW:[GetYear]::flag hundred_years_war was cleared:flavor_hyw.14"
		set_global_flag = hundred_years_war_ENG_win
		white_peace = ENG
		if = {
			limit = { singleplayer_trigger = yes }
			inherit = DAU
			change_tag = DAU
			DAU = { set_country_flag = true_france }
		}
		if = {
			limit = {
				NOT = { has_country_modifier = title_5 }
				NOT = { has_country_modifier = title_6 }
			}
			change_title_5 = yes
		}
		#_#_log = "FE-TempHYW:[GetYear]:[Root.GetName]:lost control over Paris and Northern France:flavor_hyw.14"
	}
	
	option = {				#I made France very likely to give up to reduce randomness for the player if playing England
		name = "OPT.FIGHT"
		ai_chance = { factor = 5 }
		ENG = { country_event = { id = flavor_hyw.07 days = 1 } }
		#_#_log = "FE-TempHYW:[GetYear]:[Root.GetName]:got crushed, but HYW continued:flavor_hyw.14"
		custom_tooltip = hyw_ending_explanation
	}
	
	option = {
		name = "OPT.YIELD"
		ai_chance = { factor = 95 }
		ENG = { country_event = { id = flavor_hyw.08 days = 1 } }
		#_#_log = "FE-TempHYW:[GetYear]:[Root.GetName]:got crushed, HYW ended:flavor_hyw.14"
		custom_tooltip = hyw_ending_explanation
	}
}
