levy_urban_no = { 
	cost = 0
	time = 0
	potential = { 
		ai = no
	}
	build_trigger = { 
		always = no
	}
	keep_trigger = { 
		has_building = levy_urban_no
	}
	modifier = { }
	ai_will_do = { 
		factor = 0
	}
	on_destroyed = { 
		if = { 
			limit = { 
				NOT = { 
					has_building = levy_urban_low
					has_building = levy_urban_med
					has_building = levy_urban_high
				}
			}
			add_building = levy_urban_no
		}
	}
}
