# Tax - Toll #
##############

tax_toll_no = {
	cost = 0
	time =  0
	
	potential = {
		ai = no
	}
	
	build_trigger = {
		always = no
	}
	keep_trigger = {
		has_building = tax_toll_no
	}
	
	modifier = {
	}
	
	ai_will_do = {
		factor = 0
	}

	on_destroyed = {
		if = {
			limit = {
				NOT = {
					has_building = tax_toll_low
					has_building = tax_toll_med
					has_building = tax_toll_high
				}
			}
			add_building = tax_toll_no
		}
	}
}
tax_toll_low = {
	cost = 0
	time = 0
	
	potential = {
		always = yes
	}
	
	build_trigger = {
		is_core = ROOT
		owner = {
			Tax_TaxTollCan = yes
		}
		
		NOT = {
			has_building = tax_toll_med
			has_building = tax_toll_high
		}
	}
	keep_trigger = {
		is_core = ROOT
		owner = {
			Tax_TaxTollCan = yes
		}
	}
	
	ai_will_do = {
		factor = 0
	}
	
	on_built = {
		remove_building = tax_toll_no
	}
	
	on_destroyed = {
		if = {
			limit = {
				NOT = {
					has_building = tax_toll_med
					has_building = tax_toll_high
				}
			}
			add_building = tax_toll_no
		}
	}
}
tax_toll_med = {
	cost = 0
	time = 0
	
	potential = {
		always = yes
	}
	
	build_trigger = {
		is_core = ROOT
		owner = {
			Tax_TaxTollCan = yes
		}
		
		NOT = {
			has_building = tax_toll_high
		}
	}
	keep_trigger = {
		is_core = ROOT
		owner = {
			Tax_TaxTollCan = yes
		}
	}
	
	ai_will_do = {
		factor = 0
	}
	
	on_built = {
		remove_building = tax_toll_no
        remove_building = tax_toll_low
	}
	
	on_destroyed = {
		if = {
			limit = {
				NOT = {
					has_building = tax_toll_high
				}
			}
			add_building = tax_toll_low
		}
	}
}
tax_toll_high = {
	cost = 0
	time = 0
	
	potential = {
		always = yes
	}
	
	build_trigger = {
		is_core = ROOT
		owner = {
			Tax_TaxTollCan = yes
		}
	}
	keep_trigger = {
		is_core = ROOT
		owner = {
			Tax_TaxTollCan = yes
		}
	}
	
	ai_will_do = {
		factor = 0
	}
	
	on_built = {
		remove_building = tax_toll_no
        remove_building = tax_toll_low
        remove_building = tax_toll_med
	}
	
	on_destroyed = {
		add_building = tax_toll_med
	}
}
