country_event = {

			limit = {

			}

		}
		if = {
			limit = {
				NOT = { primary_culture = low_saxon }
			}
			change_primary_culture = low_saxon
		}
		every_owned_province = {
			limit = {
				area = lusatia_area
			}
			if = {
				limit = {
					NOT = { is_core = BOH }
				}
				add_core = BOH
			}
			cede_province = BOH
		}
		2834 = { # Walcz
			if = {
				limit = {
					NOT = { is_core = POL }
				}
				add_core = POL
			}
			remove_core = BRA
			cede_province = POL
			add_permanent_province_modifier = {
				name = polish_estates
				duration = -1
			}
		}
		set_country_flag = von_luxemburg_brandenburg
		add_country_modifier = {
			name = luxemburg_rule
			duration = -1
		}
		kill_heir = yes
		define_ruler = {
			dynasty = "von Luxemburg"
			culture = czech
			adm = 1
			dip = 1
			mil = 1
			claim = 75
		}
		define_heir = {
			dynasty = "von Luxemburg"
			culture = czech
			adm = 1
			dip = 1
			mil = 1
			claim = 75
		}
		hidden_effect = {
			BOH = { set_country_flag = sale_of_the_mark }
		}
		BOH = { 
			country_event = { id = flavor_brapru.600 }
		}
		POL = {
			country_event = { id = flavor_brapru.601 }
		}
		#TODO: money should go to von Wittelsbach in some form?
	}
	
	option = {
		name = "flavor_brapru.5.b"
		ai_chance = { factor = 0 }
		capital_scope = {
			add_province_modifier = {
				name = "pretender_organizing"
				duration = 3650
			}
			add_unrest = 10
		}
		custom_tooltip = pretender_organizing_tt
		BOH = {
		
			if = { #minimum cost of 200
				limit = {
					monthly_income	= 17
				}
				add_years_of_income = 1
			}
			else = {
				add_treasury = 200
			}
			
			country_event = { id = flavor_brapru.700 }
		}
	}
}
