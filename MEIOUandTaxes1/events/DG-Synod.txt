#Synod on Protestantism
country_event = {
	id = DG_Synod.201
	title = "DG_Synod.201.title"
	desc = "DG_Synod.201.desc"
	picture = all_church_state

	is_triggered_only = yes

	immediate = {
		hidden_effect = {
			set_key = { lhs = synodvote value = 0 }
			set_key = { lhs = playervote value = 0 }
		}
		log = "DG:[GetYear]:[pro_synod_site.aab.GetValue]:Lutheran:Synod Convenes:[pro_synod_site.GetName]:[This.GetName]"
	}
	option = {
		name = "DG_Synod.201.optb"
		custom_tooltip = pro_synod_001_b
		hidden_effect = {
			every_country = {
				limit = {
					religion = catholic
					OR = {
						government = theocracy
						is_emperor = yes
						is_elector = yes
						num_of_cities = 5
						any_owned_province = { OR = { has_cardinal = yes has_province_flag = MN_Archbishopric } }
					}
				}
				if = {
					limit = { ai = no }
					country_event = { id = DG_Synod.202 }
				}
				else = {
					country_event = { id = DG_Synod.203 days = 100 }
				}
			}
			PAP = {
				country_event = { id = DG_Synod.204 days = 200 }
			}
		}
	}
	option = {
		name = "DG_Synod.201.opta"
		trigger = { ai = no }
		custom_tooltip = pro_synod_001_a
		hidden_effect = {
			set_country_flag = PAP_condemn_02
			every_country = {
				limit = { has_country_flag = protestant_presence }
				country_event = { id = DG_Protestant.005 }
			}
		}
	}
	option = {
		name = "DG_Synod.201.optc"
		trigger = { ai = no }
		Stab_Subtract4 = yes
		custom_tooltip = SetProtestantCompromiseAspects_tt
		hidden_effect = {
			SetProtestantCompromiseAspects = yes
			set_global_flag = pro_schism_compromise
			clr_global_flag = lol_schism_mended
			clr_global_flag = lol_schism_compromise
			clr_global_flag = hus_schism_mended
			clr_global_flag = hus_schism_compromise
			every_country = {
				limit = { religion = catholic }
				country_event = { id = DG_Protestant.021 }
			}
			every_country = {
				limit = { religion = protestant }
				country_event = { id = DG_Protestant.031 }
			}
		}
	}
}
