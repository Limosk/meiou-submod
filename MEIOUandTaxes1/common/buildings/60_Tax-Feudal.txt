# Tax - Feudal #
################

obli_feudal_no = {
	cost = 0
	time =  0
	
	potential = {
		ai = no
	}
	
	build_trigger = {
		always = no
	}
	keep_trigger = {
		has_building = obli_feudal_no
	}
	
	modifier = {
	}
	
	ai_will_do = {
		factor = 0
	}

	on_destroyed = {
		if = {
			limit = {
				NOT = {
					has_building = obli_feudal_low
					has_building = obli_feudal_med
					has_building = obli_feudal_high
				}
			}
			add_building = obli_feudal_no
		}
	}
}
obli_feudal_low = {
	cost = 0
	time = 0
	
	potential = {
		always = yes
	}
	
	build_trigger = {
		is_core = ROOT
		owner = {
			Tax_TaxFeudalCan = yes 
		}

		NOT = {
			has_building = obli_feudal_med
			has_building = obli_feudal_high
		}
	}
	keep_trigger = {
		is_core = ROOT
		owner = {
			Tax_TaxFeudalCan = yes 
		}
	}
	
	ai_will_do = {
		factor = 0
	}
	
	on_built = {
		remove_building = obli_feudal_no
	}
	
	on_destroyed = {
		if = {
			limit = {
				NOT = {
					has_building = obli_feudal_med
					has_building = obli_feudal_high
				}
			}
			add_building = obli_feudal_no
		}
	}
}
obli_feudal_med = {
	cost = 0
	time = 0
	
	potential = {
		always = yes
	}
	
	build_trigger = {
		is_core = ROOT
		owner = {
			Tax_TaxFeudalCan = yes 
		}

		NOT = {
			has_building = obli_feudal_high
		}
	}
	keep_trigger = {
		is_core = ROOT
		owner = {
			Tax_TaxFeudalCan = yes 
		}
	}
	
	ai_will_do = {
		factor = 0
	}
	
	on_built = {
		remove_building = obli_feudal_no
        remove_building = obli_feudal_low
	}
	
	on_destroyed = {
		if = {
			limit = {
				NOT = {
					has_building = obli_feudal_high
				}
			}
			add_building = obli_feudal_low
		}
	}
}
obli_feudal_high = {
	cost = 0
	time = 0
	
	potential = {
		always = yes
	}
	
	build_trigger = {
		is_core = ROOT
		owner = {
			Tax_TaxFeudalCan = yes 
		}
	}
	keep_trigger = {
		is_core = ROOT
		owner = {
			Tax_TaxFeudalCan = yes 
		}
	}
	
	ai_will_do = {
		factor = 0
	}
	
	on_built = {
		remove_building = obli_feudal_no
        remove_building = obli_feudal_low
        remove_building = obli_feudal_med
	}
	
	on_destroyed = {
		add_building = obli_feudal_med
	}
}
