polish_missions_main_slot = {
	slot = 1
	generic = no
	ai = yes
	potential = {
		has_country_flag = k_of_pol
	}
	has_country_shield = yes
	
	pol_restore_podolia = {
		icon = mission_eastern_european_city
		required_missions = {  }
		position = 1
		provinces_to_highlight = {
			OR = {
				area = red_ruthenia_area
				area = podolia_volhynia_area
			}
		NOT = {country_or_non_sovereign_subject_holds = ROOT}
		}
		trigger = {
			red_ruthenia_area = {
				type = all
				country_or_non_sovereign_subject_holds = ROOT
			}
			podolia_volhynia_area = {
				type = all
				country_or_non_sovereign_subject_holds = ROOT
			}
		}
		effect = {
			add_adm_power = 50
		}
	}
	
	mission_restore_ruthenia = {
		icon = mission_eastern_european_city
		required_missions = { pol_restore_podolia }
		position = 2
		provinces_to_highlight = {
			OR = {
				area = west_dniepr_area
				area = east_dniepr_area
			}
		}
		trigger = {
			west_dniepr_area = {
				type = all
				owned_by = ROOT
			}
			east_dniepr_area = {
				type = all
				owned_by = ROOT
			}
			280 = {
				base_production = 10
			}
			num_of_owned_provinces_with = {
				OR = {
					area = west_dniepr_area
					area = east_dniepr_area
				}
				base_production = 5
				value = 5
			}
		}
		effect = {
			west_dniepr_area = {
				# dev cost
				add_province_modifier = {
					name = "restored_ruthenia"
					duration = 9125 #25 years - was -1
				}
			}
			east_dniepr_area = {
				# dev cost
				add_province_modifier = {
					name = "restored_ruthenia"
					duration = 9125 #25 years - was -1
				}
			}
			crimea_region = {
				limit = {
					NOT = { country_or_non_sovereign_subject_holds = ROOT	}
					NOT = { is_core = ROOT	}
				}
				add_permanent_claim = ROOT
			}
		}
	}
	
	pol_conquer_crimea = {
		icon = mission_crusade_for_varna
		required_missions = { mission_restore_ruthenia }
		position = 3
		provinces_to_highlight = {
			OR = {
				region = crimea_region
			}
			NOT = {
				country_or_non_sovereign_subject_holds = ROOT
			}
		}
		trigger = {
			zaporizhia_area = {
				type = all
				country_or_non_sovereign_subject_holds = ROOT
			}
			sloboda_ukraine_area = {
				type = all
				country_or_non_sovereign_subject_holds = ROOT
			}
			crimea_area = {
				type = all
				country_or_non_sovereign_subject_holds = ROOT
			}
			azov_area = {
				type = all
				country_or_non_sovereign_subject_holds = ROOT
			}
		}
		effect = {
			285 = {
				add_trade_modifier = {
					who = root
					duration = 9125 #25 years - was 7300
					power = 50
					key = crimean_trade
				}
			}
		}
	}
	
	break_the_rus_mission = {
		icon = mission_conquer_5_states
		required_missions = { pol_conquer_crimea }
		position = 4
		provinces_to_highlight = {

	slot = 2
	generic = no
	ai = yes

	has_country_shield = yes
	
	lithuanian_alliance = { #mission about being able to push against TEU, achievable via alliance/vassalization of LIT, growth of miltary power without the alliance
		icon = mission_alliances
		required_missions = { }
		position = 1
		trigger = {
			if = {
				limit = {
					exists = LIT
					NOT = { is_rival = LIT }
				}
				OR = {
					alliance_with = LIT
					LIT = {
						is_subject_of = ROOT
					}
					AND = { #added to cover the situation when POL alone is strong enough to tackle TEU
						army_strength = {
							who = TEU
							value = 2 #as TEU subjects will probably double their army strength
						}
						army_size = 18
					}
				}
			}
			else = {
				lithuania_area = {
					type = all
					country_or_non_sovereign_subject_holds = ROOT
				}
			}
		}
		effect = {
			west_prussia_area = {
				limit = {
					NOT = { owned_by = ROOT	}
					NOT = { is_core = ROOT	}
				}
				add_permanent_claim = ROOT
			}
		}
	}
	pol_port_in_baltic = { #Having any port in Baltics, either via conquest in Prussia or in Pomerania
